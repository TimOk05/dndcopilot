```php
// 1. Retry логика
private function makeApiRequestWithRetry($url, $maxRetries = 3) {
    for ($i = 0; $i < $maxRetries; $i++) {
        try {
            $response = $this->makeApiRequest($url);
            if ($response) return $response;
        } catch (Exception $e) {
            if ($i === $maxRetries - 1) throw $e;
            sleep(pow(2, $i)); // Exponential backoff
        }
    }
}

// 2. Показ ошибки (БЕЗ fallback!)
if (!$apiResponse) {
    return [
        'success' => false,
        'error' => 'API недоступен',
        'message' => 'Не удалось получить данные от внешнего API',
        'details' => 'Проверьте подключение к интернету и настройки'
    ];
}
```
